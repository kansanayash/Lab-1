MAIN JAVA

public class Main
{
	public static void main(String[] args) {
		//System.out.println("Hello World");
		Mother m = new Mother() ;
		m.show()  ; //first function without argument
		m.show(3) ; //second function with   argument
		Child c = new Child() ;
		c.show() ;
		
		Mother m1 = new Child(); //run time binding or polymorphism + content of object is 'child' 
		m1.show() ;
		
		Mother m2 = new Mother(6) ;
		
	}
}




CHILD JAVA 

class Child extends Mother{
    
    public Child(){
        super(5); // calls the constructor of super class
    }
    public void show(){
        System.out.println("I am Child of Mother..") ;
    }
}



MOTHER JAVA

class Mother{
    int x ; 
    public Mother(int x){    //constructor declaration  
        this.x = 3 ;
    }
    public void show(){
        System.out.println("I am Mother..") ;
    }
    
    public void show(int a){
        System.out.println("second function : "+a);
    }
}
